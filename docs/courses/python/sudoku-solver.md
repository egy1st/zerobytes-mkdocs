 def solveSudoku(board) -> None:
        
        """
        Do not return anything, modify board in-place instead.
        """
        h = 1
        all = 0 
        old_all = 0
        con = True
       
        while (con):
                        
      
            columns = list(zip(*board))
            box = [[] for _ in range(9)]
            two = []

            for x in range(9):
                for y in range (9):
                    num = board[x][y]
                    if num != '.' :
                        idx = (x // 3) * 3 + y // 3
                        box[idx].append(num)



            for x in range(9):
                for y in range (9):
                    num = board[x][y]
                    if num == '.' or type(num) == list:
                        mylist = []

                        row = board[x]
                        col = columns[y]
                        idx = (x // 3) * 3 + (y // 3)

                        #print (num, x, y)

                        for n in range(9):


                            str_n = str(n+1)
                            if str_n not in col and str_n not in row and str_n not in box[idx]:
                                mylist.append(str_n)

                        board[x][y] = mylist


         
            old_all = all
            for x in range(9):
                for y in range (9):
                    num = board[x][y] 
                    if type(num) == list:
                        if (len(num) == 1):
                            board[x][y] = num[0]
                            all += 1
                        elif (len(num) == 2):  
                            two = num
                            pos = []
                            pos.append(x)
                            pos.append(y)
                           
                            
            if all == old_all:
                if len(two) == 2:
                    print(pos, two)

                else:
                    con = False
                    for g in range(9):

            print (h, all)

        return board
